Ruby — Rouge syntax highlight

{% highlight ruby %} # Zer0-Mistakes skin helper module Zer0 SKINS = %w[aqua neon carbon forest rose].freeze def self.apply_skin(name) raise ArgumentError, "Unknown skin: #{name}" unless SKINS.include?(name) document.documentElement.setAttribute('data-theme-skin', name) end end {% endhighlight %}
