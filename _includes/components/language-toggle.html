{%- comment -%} Component: language-toggle Path: _includes/components/language-toggle.html Purpose: Navbar dropdown that switches between the English original and its machine-generated translations (scripts/translate.rb). Shows the source language plus every configured target language; targets without a generated translation for the current page render as disabled entries. Params: none — reads page.lang / page.url / page.translation_source_url and the generated _data/i18n/manifest.yml. Depends on: site.translation.* config, site.data.i18n.manifest, _data/i18n/languages.yml (display names), core/i18n.html, Bootstrap 5 dropdown, Bootstrap Icons (bi-translate) Notes: Renders nothing when translation is disabled or no target languages are configured. The chosen language is remembered in localStorage("zer0-lang") for future UX (no auto-redirect). Must NOT be include_cached — the menu is page-dependent. {%- endcomment -%} {%- assign _lt_langs = site.translation.languages -%} {%- if site.translation.enabled and _lt_langs and _lt_langs.size > 0 -%} {%- include core/i18n.html -%} {%- assign _lt_source = site.translation.source_lang | default: 'en' -%} {%- assign _lt_current = page.lang | default: _lt_source -%} {%- if _lt_current == _lt_source -%} {%- assign _lt_src_url = page.url -%} {%- else -%} {%- assign _lt_src_url = page.translation_source_url -%} {%- endif -%} {%- assign _lt_entry = site.data.i18n.manifest.pages[_lt_src_url] -%} {%- assign _lt_names = site.data.i18n.languages -%} {%- assign _lt_source_name = _lt_names[_lt_source].native | default: _lt_source -%}
  • {%- if _lt_current == _lt_source %} {{ _lt_source_name }} {%- else %} {{ _lt_source_name }} {%- endif %}
    • {%- for _lt_lang in _lt_langs -%} {%- assign _lt_name = _lt_names[_lt_lang].native | default: _lt_lang -%} {%- assign _lt_t = _lt_entry[_lt_lang] %}
  • {%- if _lt_current == _lt_lang %} {{ _lt_name }} {%- elsif _lt_t %} {{ _lt_name }} {%- else %} {{ _lt_name }} ({{ ui.lang_not_available | default: 'Not yet translated' }}) {%- endif %}
    • {%- endfor %}
{%- endif -%}
